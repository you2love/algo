排序算法

什么是排序算法？

排序算法是将一组数据按特定顺序排列的算法。常见的排序算法包括冒泡排序、选择排序、插入排序、快速排序、归并排序等。

时间复杂度

O(n²) - O(n log n)

空间复杂度

O(1) - O(n)

稳定性

部分稳定

冒泡排序 (Bubble Sort)

重复遍历数组，比较相邻元素，如果顺序错误就交换。每一轮遍历都会将最大的元素"冒泡"到正确的位置。


function bubbleSort(arr) {
    const n = arr.length;
    for (let i = 0; i < n - 1; i++) {
        for (let j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                [arr[j], arr[j + 1]] = [arr[j + 1], arr[j]];
            }
        }
    }
    return arr;
}

快速排序 (Quick Sort)

选择基准值，将数组分为两部分：小于基准值的元素和大于基准值的元素，然后递归地对这两部分进行排序。


function quickSort(arr, left = 0, right = arr.length - 1) {
    if (left < right) {
        const pivotIndex = partition(arr, left, right);
        quickSort(arr, left, pivotIndex - 1);
        quickSort(arr, pivotIndex + 1, right);
    }
    return arr;
}

function partition(arr, left, right) {
    const pivot = arr[right];
    let i = left - 1;
    
    for (let j = left; j < right; j++) {
        if (arr[j] <= pivot) {
            i++;
            [arr[i], arr[j]] = [arr[j], arr[i]];
        }
    }
    
    [arr[i + 1], arr[right]] = [arr[right], arr[i + 1]];
    return i + 1;
}

    pivot := arr[right]
    i := left - 1

    for j := left; j < right; j++ {
        if arr[j] <= pivot {
            i++
            arr[i], arr[j] = arr[j], arr[i]
        }
    }

    arr[i+1], arr[right] = arr[right], arr[i+1]
    return i + 1
}

归并排序 (Merge Sort)

采用分治策略，将数组分成两半，分别排序后合并。是一种稳定的排序算法，时间复杂度为O(n log n)。


function mergeSort(arr) {
    if (arr.length <= 1) return arr;
    
    const mid = Math.floor(arr.length / 2);
    const left = mergeSort(arr.slice(0, mid));
    const right = mergeSort(arr.slice(mid));
    
    return merge(left, right);
}

function merge(left, right) {
    const result = [];
    let i = 0, j = 0;
    
    while (i < left.length && j < right.length) {
        if (left[i] <= right[j]) {
            result.push(left[i++]);
        } else {
            result.push(right[j++]);
        }
    }
    
    return result.concat(left.slice(i)).concat(right.slice(j));
}

    result = []
    i = j = 0

        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    result.extend(left[i:])
    result.extend(right[j:])

func mergeSort(arr []int) []int {
    if len(arr) <= 1 {
        return arr
    }

    mid := len(arr) / 2
    left := mergeSort(arr[:mid])
    right := mergeSort(arr[mid:])

    return merge(left, right)
}

func merge(left, right []int) []int {
    result := make([]int, 0, len(left)+len(right))
    i, j := 0, 0

    for i < len(left) && j < len(right) {
        if left[i] <= right[j] {
            result = append(result, left[i])
            i++
        } else {
            result = append(result, right[j])
            j++
        }
    }

    result = append(result, left[i:]...)
    result = append(result, right[j:]...)
    return result
}

    sorted := mergeSort(arr)
    fmt.Println(sorted) // [11 12 22 25 34 64 90]
}

堆排序 (Heap Sort)

利用堆这种数据结构设计的排序算法。堆是一个近似完全二叉树的结构，分为大顶堆和小顶堆。


function heapSort(arr) {
    const n = arr.length;
    
    // 构建最大堆
    for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    
    // 逐个提取元素
    for (let i = n - 1; i > 0; i--) {
        [arr[0], arr[i]] = [arr[i], arr[0]];
        heapify(arr, i, 0);
    }
    
    return arr;
}

function heapify(arr, n, i) {
    let largest = i;
    const left = 2 * i + 1;
    const right = 2 * i + 2;
    
    if (left < n && arr[left] > arr[largest]) {
        largest = left;
    }
    
    if (right < n && arr[right] > arr[largest]) {
        largest = right;
    }
    
    if (largest !== i) {
        [arr[i], arr[largest]] = [arr[largest], arr[i]];
        heapify(arr, n, largest);
    }
}

计数排序 (Counting Sort)

非比较型排序算法，通过统计每个值出现的次数来确定其位置。适用于整数范围较小的场景。


function countingSort(arr) {
    if (arr.length === 0) return arr;
    
    const min = Math.min(...arr);
    const max = Math.max(...arr);
    const range = max - min + 1;
    
    const count = new Array(range).fill(0);
    const output = new Array(arr.length);
    
    // 统计每个元素出现次数
    for (let i = 0; i < arr.length; i++) {
        count[arr[i] - min]++;
    }
    
    // 计算累积计数
    for (let i = 1; i < count.length; i++) {
        count[i] += count[i - 1];
    }
    
    // 构建输出数组
    for (let i = arr.length - 1; i >= 0; i--) {
        output[count[arr[i] - min] - 1] = arr[i];
        count[arr[i] - min]--;
    }
    
    return output;
}

def counting_sort(arr):
    if not arr:
        return arr
    
    min_val = min(arr)
    max_val = max(arr)
    range_val = max_val - min_val + 1
    
    count = [0] * range_val
    output = [0] * len(arr)
    
    # 统计每个元素出现次数
    for num in arr:
        count[num - min_val] += 1
    
    # 计算累积计数
    for i in range(1, len(count)):
        count[i] += count[i - 1]
    
    # 构建输出数组
    for i in range(len(arr) - 1, -1, -1):
        output[count[arr[i] - min_val] - 1] = arr[i]
        count[arr[i] - min_val] -= 1
    
    return output

func countingSort(arr []int) []int {
    if len(arr) == 0 {
        return arr
    }

    // 找到最小值和最大值
    minVal, maxVal := arr[0], arr[0]
    for _, num := range arr {
        if num < minVal {
            minVal = num
        }
        if num > maxVal {
            maxVal = num
        }
    }

    rangeVal := maxVal - minVal + 1
    count := make([]int, rangeVal)
    output := make([]int, len(arr))

        count[num-minVal]++
    }

    for i := 1; i < len(count); i++ {
        count[i] += count[i-1]
    }

    for i := len(arr) - 1; i >= 0; i-- {
        output[count[arr[i]-minVal]-1] = arr[i]
        count[arr[i]-minVal]--
    }

    return output
}

桶排序 (Bucket Sort)

将数组分到有限数量的桶里，每个桶再分别排序。适用于数据均匀分布的场景。


function bucketSort(arr, bucketSize = 5) {
    if (arr.length === 0) return arr;
    
    const min = Math.min(...arr);
    const max = Math.max(...arr);
    
    const bucketCount = Math.floor((max - min) / bucketSize) + 1;
    const buckets = new Array(bucketCount).fill(null).map(() => []);
    
    // 分配到桶中
    for (let i = 0; i < arr.length; i++) {
        const bucketIndex = Math.floor((arr[i] - min) / bucketSize);
        buckets[bucketIndex].push(arr[i]);
    }
    
    // 每个桶内排序
    const result = [];
    for (let i = 0; i < buckets.length; i++) {
        if (buckets[i].length > 0) {
            buckets[i].sort((a, b) => a - b);
            result.push(...buckets[i]);
        }
    }
    
    return result;
}

def bucket_sort(arr, bucket_size=5):
    if not arr:
        return arr
    
    min_val = min(arr)
    max_val = max(arr)
    
    bucket_count = (max_val - min_val) // bucket_size + 1
    buckets = [[] for _ in range(bucket_count)]
    
    # 分配到桶中
    for num in arr:
        bucket_index = (num - min_val) // bucket_size
        buckets[bucket_index].append(num)
    
    # 每个桶内排序
    result = []
    for bucket in buckets:
        if bucket:
            bucket.sort()
            result.extend(bucket)
    
    return result

package main

import (
    "fmt"
    "sort"
)

func bucketSort(arr []int, bucketSize int) []int {
    if len(arr) == 0 {
        return arr
    }

    // 找到最小值和最大值
    minVal, maxVal := arr[0], arr[0]
    for _, num := range arr {
        if num < minVal {
            minVal = num
        }
        if num > maxVal {
            maxVal = num
        }
    }

    // 计算桶的数量
    bucketCount := (maxVal-minVal)/bucketSize + 1
    buckets := make([][]int, bucketCount)
    for i := range buckets {
        buckets[i] = make([]int, 0)
    }

    // 分配到桶中
    for _, num := range arr {
        bucketIndex := (num - minVal) / bucketSize
        buckets[bucketIndex] = append(buckets[bucketIndex], num)
    }

    // 每个桶内排序并合并结果
    result := make([]int, 0, len(arr))
    for _, bucket := range buckets {
        if len(bucket) > 0 {
            sort.Ints(bucket)
            result = append(result, bucket...)
        }
    }

    return result
}

func main() {
    arr := []int{29, 25, 3, 49, 9, 37, 21, 43}
    sorted := bucketSort(arr, 5)
    fmt.Println(sorted) // [3 9 21 25 29 37 43 49]
}

AI 时代的排序算法

神经网络辅助排序 (Neural Sort)

利用神经网络学习数据的分布特征，预测最优排序策略。特别适用于部分有序或具有特定模式的数据。


// 简化的神经网络辅助排序概念实现
class NeuralSort {
    constructor() {
        this.learnedPatterns = new Map();
    }
    
    // 分析数据特征
    analyzePattern(arr) {
        const n = arr.length;
        let sortedCount = 0;
        let reverseCount = 0;
        
        for (let i = 1; i < n; i++) {
            if (arr[i] >= arr[i-1]) sortedCount++;
            if (arr[i] <= arr[i-1]) reverseCount++;
        }
        
        return {
            isNearlySorted: sortedCount >= n * 0.8,
            isNearlyReverse: reverseCount >= n * 0.8,
            sortedness: sortedCount / (n - 1)
        };
    }
    
    // 根据模式选择最优排序策略
    smartSort(arr) {
        const pattern = this.analyzePattern(arr);
        
        if (pattern.isNearlySorted) {
            return this.insertionSort(arr);  // 近乎有序时用插入排序
        }
        
        if (pattern.sortedness > 0.5) {
            return this.bubbleSortOptimized(arr);  // 部分有序时用优化冒泡
        }
        
        return this.quickSort(arr);  // 默认使用快速排序
    }
    
    insertionSort(arr) {
        const result = [...arr];
        for (let i = 1; i < result.length; i++) {
            let j = i - 1;
            const key = result[i];
            while (j >= 0 && result[j] > key) {
                result[j + 1] = result[j];
                j--;
            }
            result[j + 1] = key;
        }
        return result;
    }
    
    bubbleSortOptimized(arr) {
        const result = [...arr];
        let swapped;
        do {
            swapped = false;
            for (let i = 0; i < result.length - 1; i++) {
                if (result[i] > result[i + 1]) {
                    [result[i], result[i + 1]] = [result[i + 1], result[i]];
                    swapped = true;
                }
            }
        } while (swapped);
        return result;
    }
    
    quickSort(arr) {
        if (arr.length <= 1) return arr;
        const pivot = arr[Math.floor(arr.length / 2)];
        const left = arr.filter(x => x < pivot);
        const middle = arr.filter(x => x === pivot);
        const right = arr.filter(x => x > pivot);
        return [...this.quickSort(left), ...middle, ...this.quickSort(right)];
    }
}

import numpy as np
from typing import List

class NeuralSort:
    def __init__(self):
        self.learned_patterns = {}
    
    def analyze_pattern(self, arr: List[int]) -> dict:
        """分析数据模式"""
        n = len(arr)
        if n <= 1:
            return {'is_nearly_sorted': True, 'sortedness': 1.0}
        
        sorted_count = sum(1 for i in range(1, n) if arr[i] >= arr[i-1])
        reverse_count = sum(1 for i in range(1, n) if arr[i] <= arr[i-1])
        
        return {
            'is_nearly_sorted': sorted_count >= n * 0.8,
            'is_nearly_reverse': reverse_count >= n * 0.8,
            'sortedness': sorted_count / (n - 1)
        }
    
    def smart_sort(self, arr: List[int]) -> List[int]:
        """根据模式选择最优排序策略"""
        pattern = self.analyze_pattern(arr)
        
        if pattern['is_nearly_sorted']:
            return self._insertion_sort(arr)
        
        if pattern['sortedness'] > 0.5:
            return self._bubble_sort_optimized(arr)
        
        return self._quick_sort(arr)
    
    def _insertion_sort(self, arr: List[int]) -> List[int]:
        result = arr.copy()
        for i in range(1, len(result)):
            key = result[i]
            j = i - 1
            while j >= 0 and result[j] > key:
                result[j + 1] = result[j]
                j -= 1
            result[j + 1] = key
        return result
    
    def _bubble_sort_optimized(self, arr: List[int]) -> List[int]:
        result = arr.copy()
        while True:
            swapped = False
            for i in range(len(result) - 1):
                if result[i] > result[i + 1]:
                    result[i], result[i + 1] = result[i + 1], result[i]
                    swapped = True
            if not swapped:
                break
        return result
    
    def _quick_sort(self, arr: List[int]) -> List[int]:
        if len(arr) <= 1:
            return arr
        pivot = arr[len(arr) // 2]
        left = [x for x in arr if x < pivot]
        middle = [x for x in arr if x == pivot]
        right = [x for x in arr if x > pivot]
        return self._quick_sort(left) + middle + self._quick_sort(right)

        }
    }

    return Pattern{
        IsNearlySorted: float64(sortedCount) >= float64(n)*0.8,
        IsNearlyReverse: float64(reverseCount) >= float64(n)*0.8,
        Sortedness:     float64(sortedCount) / float64(n-1),
    }
}

// SmartSort 根据模式选择最优排序策略
func (ns *NeuralSort) SmartSort(arr []int) []int {
    pattern := ns.AnalyzePattern(arr)

    if pattern.IsNearlySorted {
        return ns.insertionSort(arr) // 近乎有序时用插入排序
    }

    if pattern.Sortedness > 0.5 {
        return ns.bubbleSortOptimized(arr) // 部分有序时用优化冒泡
    }

    return ns.quickSort(arr) // 默认使用快速排序
}

// insertionSort 插入排序
func (ns *NeuralSort) insertionSort(arr []int) []int {
    result := make([]int, len(arr))
    copy(result, arr)

    for i := 1; i < len(result); i++ {
        key := result[i]
        j := i - 1
        for j >= 0 && result[j] > key {
            result[j+1] = result[j]
            j--
        }
        result[j+1] = key
    }
    return result
}

// bubbleSortOptimized 优化的冒泡排序
func (ns *NeuralSort) bubbleSortOptimized(arr []int) []int {
    result := make([]int, len(arr))
    copy(result, arr)

    for {
        swapped := false
        for i := 0; i < len(result)-1; i++ {
            if result[i] > result[i+1] {
                result[i], result[i+1] = result[i+1], result[i]
                swapped = true
            }
        }
        if !swapped {
            break
        }
    }
    return result
}

// quickSort 快速排序
func (ns *NeuralSort) quickSort(arr []int) []int {
    if len(arr) <= 1 {
        return arr
    }

    pivot := arr[len(arr)/2]
    left := make([]int, 0)
    middle := make([]int, 0)
    right := make([]int, 0)

    for _, x := range arr {
        if x < pivot {
            left = append(left, x)
        } else if x == pivot {
            middle = append(middle, x)
        } else {
            right = append(right, x)
        }
    }

    left = ns.quickSort(left)
    right = ns.quickSort(right)

    result := make([]int, 0, len(left)+len(middle)+len(right))
    result = append(result, left...)
    result = append(result, middle...)
    result = append(result, right...)
    return result
}

    ns := NewNeuralSort()
    fmt.Println(ns.SmartSort([]int{1, 2, 3, 5, 4, 6, 7}))     // 近乎有序
    fmt.Println(ns.SmartSort([]int{64, 34, 25, 12, 22, 11, 90})) // 随机
}

并行排序 (Parallel Sort)

利用多核处理器并行计算能力，将数据分块后同时排序，最后合并结果。适用于大数据集。


// 使用 Web Workers 的并行排序（浏览器环境）
class ParallelSort {
    constructor(numWorkers = 4) {
        this.numWorkers = numWorkers;
    }
    
    async sort(arr) {
        if (arr.length < 1000) {
            return arr.sort((a, b) => a - b);
        }
        
        const chunkSize = Math.ceil(arr.length / this.numWorkers);
        const chunks = [];
        
        for (let i = 0; i < arr.length; i += chunkSize) {
            chunks.push(arr.slice(i, i + chunkSize));
        }
        
        // 并行排序每个块
        const sortedChunks = await Promise.all(
            chunks.map(chunk => this._sortChunk(chunk))
        );
        
        // 合并结果
        return this._mergeAll(sortedChunks);
    }
    
    _sortChunk(chunk) {
        return new Promise(resolve => {
            setTimeout(() => resolve(chunk.sort((a, b) => a - b)), 0);
        });
    }
    
    _mergeAll(chunks) {
        while (chunks.length > 1) {
            const merged = [];
            for (let i = 0; i < chunks.length; i += 2) {
                if (i + 1 < chunks.length) {
                    merged.push(this._merge(chunks[i], chunks[i + 1]));
                } else {
                    merged.push(chunks[i]);
                }
            }
            chunks = merged;
        }
        return chunks[0];
    }
    
    _merge(left, right) {
        const result = [];
        let i = 0, j = 0;
        
        while (i < left.length && j < right.length) {
            if (left[i] <= right[j]) {
                result.push(left[i++]);
            } else {
                result.push(right[j++]);
            }
        }
        
        return result.concat(left.slice(i)).concat(right.slice(j));
    }
}

from concurrent.futures import ProcessPoolExecutor
from typing import List
import multiprocessing

class ParallelSort:
    def __init__(self, num_workers: int = None):
        self.num_workers = num_workers or multiprocessing.cpu_count()
    
    def sort(self, arr: List[int]) -> List[int]:
        if len(arr) < 1000:
            return sorted(arr)
        
        chunk_size = (len(arr) + self.num_workers - 1) // self.num_workers
        chunks = [arr[i:i + chunk_size] for i in range(0, len(arr), chunk_size)]
        
        # 并行排序每个块
        with ProcessPoolExecutor(max_workers=self.num_workers) as executor:
            sorted_chunks = list(executor.map(sorted, chunks))
        
        # 合并结果
        return self._merge_all(sorted_chunks)
    
    def _merge_all(self, chunks: List[List[int]]) -> List[int]:
        while len(chunks) > 1:
            merged = []
            for i in range(0, len(chunks), 2):
                if i + 1 < len(chunks):
                    merged.append(self._merge(chunks[i], chunks[i + 1]))
                else:
                    merged.append(chunks[i])
            chunks = merged
        return chunks[0] if chunks else []
    
    def _merge(self, left: List[int], right: List[int]) -> List[int]:
        result = []
        i = j = 0
        
        while i < len(left) and j < len(right):
            if left[i] <= right[j]:
                result.append(left[i])
                i += 1
            else:
                result.append(right[j])
                j += 1
        
        result.extend(left[i:])
        result.extend(right[j:])
        return result

type ParallelSort struct {

// NewParallelSort 创建新的 ParallelSort 实例
func NewParallelSort(numWorkers int) *ParallelSort {
    if numWorkers <= 0 {
        numWorkers = 4
    }
    return &ParallelSort{numWorkers: numWorkers}
}

// Sort 并行排序主函数
func (ps *ParallelSort) Sort(arr []int) []int {
    // 小数组直接排序
    if len(arr) < 1000 {
        result := make([]int, len(arr))
        copy(result, arr)
        sort.Ints(result)
        return result
    }

    // 分割数组
    chunkSize := (len(arr) + ps.numWorkers - 1) / ps.numWorkers
    chunks := make([][]int, 0, ps.numWorkers)

    for i := 0; i < len(arr); i += chunkSize {
        end := i + chunkSize
        if end > len(arr) {
            end = len(arr)
        }
        chunk := make([]int, end-i)
        copy(chunk, arr[i:end])
        chunks = append(chunks, chunk)
    }

    // 并行排序每个块
    var wg sync.WaitGroup
    sortedChunks := make([][]int, len(chunks))

    for i, chunk := range chunks {
        wg.Add(1)
        go func(idx int, c []int) {
            defer wg.Done()
            sort.Ints(c)
            sortedChunks[idx] = c
        }(i, chunk)
    }

    wg.Wait()

    // 合并结果
    return ps.mergeAll(sortedChunks)
}

// mergeAll 合并所有已排序的块
func (ps *ParallelSort) mergeAll(chunks [][]int) []int {
    for len(chunks) > 1 {
        merged := make([][]int, 0, (len(chunks)+1)/2)
        for i := 0; i < len(chunks); i += 2 {
            if i+1 < len(chunks) {
                merged = append(merged, ps.merge(chunks[i], chunks[i+1]))
            } else {
                merged = append(merged, chunks[i])
            }
        }
        chunks = merged
    }
    if len(chunks) > 0 {
        return chunks[0]
    }
    return []int{}
}

// merge 合并两个已排序的数组
func (ps *ParallelSort) merge(left, right []int) []int {
    result := make([]int, 0, len(left)+len(right))
    i, j := 0, 0

    for i < len(left) && j < len(right) {
        if left[i] <= right[j] {
            result = append(result, left[i])
            i++
        } else {
            result = append(result, right[j])
            j++
        }
    }

    result = append(result, left[i:]...)
    result = append(result, right[j:]...)
    return result
}

量子排序概念 (Quantum Sort Concept)

基于量子计算原理的排序算法概念。利用量子叠加和量子纠缠特性，理论上可实现 O(√n) 的排序复杂度。


"""
量子排序概念实现 - 模拟量子比较网络
注意：这是经典计算机上的模拟，真正的量子排序需要量子计算机
"""
import numpy as np
from typing import List

class QuantumSortConcept:
    def __init__(self):
        self.qubits = None
    
    def _create_superposition(self, arr: List[int]) -> np.ndarray:
        """创建叠加态（模拟）"""
        n = len(arr)
        # 在真实量子计算机中，这会创建 2^n 个状态的叠加
        return np.array(arr)
    
    def _quantum_compare(self, a: int, b: int) -> tuple:
        """量子比较操作（模拟）"""
        # 真实量子计算中，这通过量子门实现
        if a <= b:
            return a, b
        else:
            return b, a
    
    def _quantum_sort_network(self, arr: List[int]) -> List[int]:
        """量子排序网络（模拟）"""
        n = len(arr)
        result = arr.copy()
        
        # 使用 Bitonic 排序网络（可并行化）
        def bitonic_merge(arr, low, cnt, direction):
            if cnt > 1:
                k = cnt // 2
                for i in range(low, low + k):
                    if (direction == 1 and arr[i] > arr[i + k]) or \
                       (direction == 0 and arr[i] < arr[i + k]):
                        arr[i], arr[i + k] = self._quantum_compare(arr[i], arr[i + k])
                bitonic_merge(arr, low, k, direction)
                bitonic_merge(arr, low + k, k, direction)
        
        def bitonic_sort(arr, low, cnt, direction):
            if cnt > 1:
                k = cnt // 2
                bitonic_sort(arr, low, k, 1)
                bitonic_sort(arr, low + k, k, 0)
                bitonic_merge(arr, low, cnt, direction)
        
        bitonic_sort(result, 0, n, 1)
        return result
    
    def sort(self, arr: List[int]) -> List[int]:
        """
        量子排序接口
        时间复杂度：O(log²n) - 使用量子并行性
        空间复杂度：O(n) - 量子比特数
        """
        if len(arr) <= 1:
            return arr
        
        # 检查长度是否为 2 的幂（量子排序网络要求）
        n = len(arr)
        if n & (n - 1) != 0:
            # 填充到最近的 2 的幂
            next_pow2 = 1
            while next_pow2 < n:
                next_pow2 *= 2
            arr = arr + [float('inf')] * (next_pow2 - n)
        
        result = self._quantum_sort_network(arr)
        
        # 移除填充元素
        return [x for x in result if x != float('inf')]

func NewQuantumSortConcept() *QuantumSortConcept {

// quantumCompare 量子比较操作（模拟）
// 真实量子计算中，这通过量子门实现
func (qs *QuantumSortConcept) quantumCompare(a, b int) (int, int) {
    if a <= b {
        return a, b
    }
    return b, a
}

// bitonicMerge Bitonic 排序网络的合并操作
func (qs *QuantumSortConcept) bitonicMerge(arr []int, low, cnt, direction int) {
    if cnt > 1 {
        k := cnt / 2
        for i := low; i < low+k; i++ {
            // 根据方向决定是否交换
            if (direction == 1 && arr[i] > arr[i+k]) ||
                (direction == 0 && arr[i] < arr[i+k]) {
                arr[i], arr[i+k] = qs.quantumCompare(arr[i], arr[i+k])
            }
        }
        qs.bitonicMerge(arr, low, k, direction)
        qs.bitonicMerge(arr, low+k, k, direction)
    }
}

// bitonicSort Bitonic 排序网络
// 这是一种可以高度并行化的排序网络，适合模拟量子并行性
func (qs *QuantumSortConcept) bitonicSort(arr []int, low, cnt, direction int) {
    if cnt > 1 {
        k := cnt / 2
        // 递归排序两半，方向相反
        qs.bitonicSort(arr, low, k, 1)   // 升序
        qs.bitonicSort(arr, low+k, k, 0) // 降序
        // 合并
        qs.bitonicMerge(arr, low, cnt, direction)
    }
}

// nextPowerOf2 计算大于等于 n 的最小的 2 的幂
func nextPowerOf2(n int) int {
    if n <= 0 {
        return 1
    }
    return int(math.Pow(2, math.Ceil(math.Log2(float64(n)))))
}

// Sort 量子排序接口
// 时间复杂度：O(log²n) - 使用量子并行性（模拟）
// 空间复杂度：O(n) - 量子比特数
func (qs *QuantumSortConcept) Sort(arr []int) []int {
    if len(arr) <= 1 {
        return arr
    }

    n := len(arr)
    // 创建副本避免修改原数组
    result := make([]int, len(arr))
    copy(result, arr)

    // 检查长度是否为 2 的幂（量子排序网络要求）
    if n&(n-1) != 0 {
        // 填充到最近的 2 的幂
        nextPow2 := nextPowerOf2(n)
        // 使用最大值填充，排序后移除
        for len(result) < nextPow2 {
            result = append(result, math.MaxInt32)
        }
    }

    // 执行 Bitonic 排序网络
    qs.bitonicSort(result, 0, len(result), 1)

    // 移除填充元素
    filtered := make([]int, 0, n)
    for _, v := range result {
        if v != math.MaxInt32 {
            filtered = append(filtered, v)
        }
    }
    return filtered
}
